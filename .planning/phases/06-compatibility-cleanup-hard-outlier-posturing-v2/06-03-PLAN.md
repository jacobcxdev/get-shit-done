    export function extractLauncherBlock(content: string): string | null {
      const trimmed = content.trim();
      // Must start with ```gsd-advisory and end with ```
      const m = trimmed.match(/^```gsd-advisory\r?\n([\s\S]+?)\r?\n```$/);
      if (!m) return null;
      // The trimmed content must be exactly the matched block — no extra text
      if (trimmed !== m[0]) return null;

    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Workflow file content → extractLauncherBlock | File content from disk is untrusted; regex must not backtrack catastrophically on large prose files |
| Launcher YAML → validateLauncherMetadata | YAML block content is untrusted; parsed via the same flat parser as posture records — no code execution |
| commandId in launcher → SEED_HARD_OUTLIERS | Launcher metadata cannot override hard-outlier classification |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-03-01 | Tampering | thin launcher commandId for hard outlier | mitigate | validateLauncherMetadata checks commandId against SEED_HARD_OUTLIERS and emits SLIM-02 — hard outliers cannot acquire a thin launcher |
| T-06-03-02 | Tampering | workflowId mismatch in launcher | mitigate | validateLauncherMetadata derives expected workflowId from filePath and rejects mismatches — a launcher cannot masquerade as a different workflow |
| T-06-03-03 | Denial of Service | catastrophic regex backtrack on large prose files | mitigate | extractLauncherBlock regex is anchored (^ and $) and uses lazy quantifier on the interior; prose files with no gsd-advisory fence will fail the initial match quickly |
| T-06-03-04 | Information Disclosure | filePath in SLIM-02 diagnostics | accept | Local development tool; file paths in compiler diagnostics are expected and useful |
| T-06-03-05 | Spoofing | fake isLauncher:true by adding a gsd-advisory block to a prose file | mitigate | extractLauncherBlock requires the block to be the ENTIRE file content (trimmed); any surrounding prose causes null return |
</threat_model>
